1. Sử dụng function in ra thông tin tên và loại một sản phẩm bất kỳ. Tạo một function để in tên và loại một sản phẩm bất kỳ cho trước. 

function formatPN(sp) {
    return sp.name + " - " + sp.category;
}

const sp = {
    name: "May khoan pin",
    category: "Dien tu"
}

function DanhThiep() {
    return(
        <h1>San pham {formatPN(sp)}!</h1>
    );
}

export default DanhThiep;
